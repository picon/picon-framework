 25: /**
 26:  * Based on SplEnum, this class allows enumerated types to be created.
 27:  * 
 28:  * Extend this class to create your enums in the following way
 29:  * 
 30:  * <code>
 31:  * class Fruit extends Enum
 32:  * {
 33:  *      const APPLE = 1;
 34:  *      const ORANGE = 2;
 35:  *      const _DEFAULT = 1;
 36:  * }
 37:  * </code>
 38:  * 
 39:  * _DEFAULT is optional and will be used if no paramater is passed to the constructor
 40:  * 
 41:  * <code>
 42:  * echo new Fruit();
 43:  * echo new Fruit(Fruit::ORANGE);
 44:  * echo Fruit::valueOf("orange");
 45:  * </code>
 46:  * 
 47:  * Will output:
 48:  * 1
 49:  * 2
 50:  * 2
 51:  * 
 52:  * @author Martin Cassidy
 53:  * @package domain
 54:  */
 55: abstract class Enum implements Identifiable, Equalable
 56: {
 57:     private $value;
 58:     const DEFAULT_NAME = "_DEFAULT";
 59: 
 60:     /**
 61:      * @param null $value
 62:      */
 63:     public function __construct($value = null)
 64:     {
 65:         $enum = new \ReflectionClass ($this);
 66:         
 67:         if($value==null)
 68:         {
 69:             if(!$enum->hasConstant(self::DEFAULT_NAME))
 70:             {
 71:                 throw new \InvalidArgumentException("Cannot create enum as no value has been passed and no default is specified");
 72:             }
 73:             else
 74:             {
 75:                 $value = $enum->getConstant(self::DEFAULT_NAME);
 76:             }
 77:         }
 78:         
 79:         foreach($enum->getConstants() as $name => $enumValue)
 80:         {
 81:             if($enumValue==$value)
 82:             {
 83:                 $this->value = $value;
 84:                 return;
 85:             }
 86:         }
 87:         
 88:         throw new \InvalidArgumentException("Unknown enum value ".$value);
 89:     }
 90:     
 91:     /**
 92:      * Create an enum for the given value
 93:      * @param string $obj A string of the value
 94:      * @return Enum The enum for the value, or null if no value can be found
 95:      */
 96:     public static function valueOf($obj)
 97:     {
 98:         $enumClass = new \ReflectionClass (get_called_class());
 99:         foreach($enumClass->getConstants() as $enumName => $enumValue)
100:         {
101:             if(strtolower($enumValue)==strtolower($obj))
102:             {
103:                 return new static($enumValue);
104:             }
105:         }
106:         return null;
107:     }
108:     
109:     /**
110:      * Gets all the posible values for this enum
111:      * 
112:      * @param boolean include the default value, defaults to false
113:      * @return array of values for this enum
114:      */
115:     public static function values($includeDefault = false)
116:     {
117:         $enumClass = new \ReflectionClass (get_called_class());
118:         $values = $enumClass->getConstants();
119:         $processed = array();
120:         
121:         foreach($values as $index => $value)
122:         {
123:             if($index!="DEFAULT_NAME" && ($includeDefault || $index!=self::DEFAULT_NAME))
124:             {
125:                 $processed[$index] = $value;
126:             }
127:         }
128:         
129:         
130:         return $processed;
131:     }
132:     
133:     /**
134:      * 
135:      * @return A String representation of the enum
136:      */
137:     public function __toString()
138:     {
139:         return "".$this->value;
140:     }
141:     
142:     /**
143:      * Tests if two enums are equal
144:      * @param Enum The enum to test against
145:      * @return Boolean true if the enum's match, false otherwise. False if the passed object is not a valid enum
146:      */
147:     public function equals($enum)
148:     {
149:         if($enum instanceof Enum)
150:         {
151:             return $enum->value==$this->value;
152:         }
153:         return false;
154:     }
155:     
156:     public static function getIdentifier()
157:     {
158:         return Identifier::forName(get_called_class());
159:     }
160: 
161: }
