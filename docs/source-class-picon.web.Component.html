  35:  * Component sersvices as the hightest and most abstract super class for all
  36:  * components. 
  37:  * 
  38:  * Component supports complex serialisation @see PiconSerializer
  39:  * Component automatically injects resources on instantiation @see Injector
  40:  * 
  41:  * A component must have a unique ID that is passed in the constructor. The ID 
  42:  * need only be unique amoung sibling component in the component hierarchy.
  43:  * 
  44:  * Components are organised into a simple hierachy. With the exception of the 
  45:  * ultimate parent at the top of the hierarchy, a component will always have a parent.
  46:  *

 299:     {
 300:         if($this->afterComponentRenderCallback!=null)
 301:         {
 302:             $callable = $this->afterComponentRender;
 303:             $callable($this);
 304:         }
 305:         $this->rendered = true;
 306:         PiconApplication::get()->getComponentAfterRenderListenersr()->onAfterRender($this);
 307:         $this->notifyBehavioursAfterRender();
 308:            
 309:         if($this instanceof MarkupContainer && $this->visible)
 310:         {
 311:             foreach($this->getChildren() as $child)
 312:             {
 313:                 /**
 314:                  * If the child component has not been rendered it must
 315:                  * not have been present in the markup
 316:                  * @todo move this into another non component parent calling method
 317:                  * so that all children may have internalAfterRender
 318:                  */
 319:                 if(!$child->rendered)
 320:                 {
 321:                     throw new \RuntimeException(sprintf("Component %s was not rendered because there was no corrisponding picon:id in the markup.", $child->id));
 322:                 }
 323:             }
 324:         }
 325:     }

 327:     public function render()
 328:     {
 329:         $exception = null;
 330:         
 331:         try
 332:         {
 333:             if($this->getParent()==null)
 334:             {
 335:                 $this->beforePageRender();
 336:             }
 337:             $this->beforeComponentRender();
 338:             $this->internalRender();
 339:         }
 340:         catch(Exception $ex)
 341:         {
 342:             $exception = $ex;
 343:         }
 344:         try
 345:         {
 346:             $this->afterComponentRender();
 347:             if($this->getParent()==null)
 348:             {
 349:                 $this->afterPageRender();
 350:             }
 351:         }
 352:         catch(Exception $ex)
 353:         {
 354:              if($exception==null)
 355:              {
 356:                 $exception = $ex;
 357:              }
 358:         }
 359:         
 360:         if($exception!=null)
 361:         {
 362:             throw $exception;
 363:         }
 364:     }

 366:     private function internalRender()
 367:     {
 368:         $markup = $this->getMarkup();
 369: 
 370:         if($markup==null)
 371:         {
 372:             throw new \MarkupNotFoundException(sprintf("Markup not found for component %s.", $this->id));
 373:         }
 374:         $this->onRender();
 375:     }
 376:     
 377:     /**
 378:      * Renders this component
 379:      */
 380:     protected final function internalRenderComponent()
 381:     {
 382:         $markup = $this->getMarkup();
 383: 
 384:         if($markup==null)
 385:         {
 386:             throw new \MarkupNotFoundException(sprintf("Markup not found for component %s.", $this->id));
 387:         }
 388:         /* @todo this cloning is a quick fix, markup should be imutable until 
 389:          * this point were a mutable version is created for use by the component 
 390:          * to render with for only this request
 391:          */
 392:         $markup = clone $markup;
 393:         
 394:         //@todo add ajax placeholder to set display:none on invisible components
 395:         if(!$this->visible)
 396:         {
 397:             return;
 398:         }
 399:         
 400:         if($this->renderBodyOnly)
 401:         {
 402:             $this->onComponentTagBody($markup);
 403:         }
 404:         else
 405:         {
 406:             $this->onComponentTag($markup);
 407:             $this->renderElementStart($markup);
 408:             $this->onComponentTagBody($markup);
 409:             $this->renderElementEnd($markup);

 462:     /**
 463:      * Render all of the markup elements in the array
 464:      * @param array An array of markup elements
 465:      */
 466:     protected function renderAll($markup = null)
 467:     {
 468:         if($markup==null)
 469:         {
 470:             $markup = $this->getMarkup();
 471: 
 472:             if($markup==null)
 473:             {
 474:                 throw new \MarkupNotFoundException(sprintf("Markup not found for component %s.", $this->id));
 475:             }
 476:         }
 477:         
 478:         foreach($markup as &$element)
 479:         {
 480:             if($element instanceof ComponentTag)
 481:             {
 482:                 if($this instanceof MarkupContainer)
 483:                 {
 484:                     $child = $this->get($element->getComponentTagId());
 485:                     
 486:                     if($child==null)
 487:                     {
 488:                         $child = ComponentResolverHelper::resolve($this, $element);
 489:                         
 490:                         if($child!=null && $child->getParent()==null)
 491:                         {
 492:                             $child->setAuto();
 493:                             $this->addComponent($child);
 494:                         }
 495:                         if($child!=null)
 496:                         {
 497:                             $child->setMarkup($element);
 498:                         }
 499:                     }
 500:                     
 501:                     if($child!=null)
 502:                     { 
 503:                         $child->render();
 504:                     }
 505:                     else
 506:                     {
 507:                         throw new \RuntimeException(sprintf("A component was not found for element with picon:id %s. This may be because you have forgotten to create it in your code or the hierarchy is wrong", $element->getComponentTagId()));
 508:                     }
 509:                     
 510:                 }
 511:                 else
 512:                 {
 513:                     throw new \InvalidMarkupException(sprintf("Markup element %s may not contain a child with a picon:id as the component %s cannot not have any child components", $element->getName(), $this->id));
 514:                 }
 515:             }
 516:             elseif($element instanceof TextElement)
 517:             {
 518:                 $this->getResponse()->write($element->getContent());
 519:             }
 520:             else
 521:             {
 522:                 $this->renderElement($element);
 523:             }
 524:         }
 525:     }
