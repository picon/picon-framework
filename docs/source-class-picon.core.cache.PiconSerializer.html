 31: /**
 32:  * A serialization helper that extends the standard PHP serialization 
 33:  * functionality by permiting serialisazation of closures and providing
 34:  * support for transient properties.
 35:  *
 36:  * @author Martin Cassidy
 37:  * @package cache
 38:  */
 39: class PiconSerializer
 40: {
 41:     private static $prepared = array();
 42:     private static $restore = array();
 43:     
 44:     private function __construct()
 45:     {
 46:         //Singleton
 47:     }
 48:     
 49:     /**
 50:      * serialize an object
 51:      * @param mixed $object
 52:      * @return string 
 53:      */
 54:     public static function serialize($object)
 55:     {
 56:         if(!is_object($object) && (!is_callable($object) && !($object instanceof \Closure)))
 57:         {
 58:             return serialize($object);
 59:         }
 60:         
 61:         if($object instanceof \Closure)
 62:         {
 63:             return serialize(new SerializableClosure($object));
 64:         }
 65:         
 66:         self::$prepared = array();
 67:         self::$restore = array();
 68:         self::prepareForSerize(new \ReflectionAnnotatedClass($object), $object);
 69:         
 70:         $serialized = serialize($object);
 71:         self::restore();
 72:         
 73:         self::$prepared = array();
 74:         self::$restore = array();
 75:         return $serialized;
 76:     }
 77:     
 78:     /**
 79:      * Unserialize an object
 80:      * @param string $string
 81:      * @return mixed 
 82:      */
 83:     public static function unserialize($string)
 84:     {
 85:         $unserialzed = unserialize($string);
 86:         
 87:         if($unserialzed instanceof InjectOnWakeup)
 88:         {
 89:             Injector::get()->inject($unserialzed);
 90:         }
 91:         
 92:         return $unserialzed;
 93:     }
 94:     
 95:     private static function prepareForSerize(\ReflectionAnnotatedClass $reflection, $object, $parent = false)
 96:     {
 97:         $hash = spl_object_hash($object);
 98:         if(in_array($hash, self::$prepared) && $parent==false)
 99:         {
100:             return;
101:         }
102:         array_push(self::$prepared, $hash);
103:         $globalAltered = false;
104:         $defaults = null;
105:         
106:         foreach($reflection->getProperties() as $property)
107:         {
108:             $altered = false;
109:             $property->setAccessible(true);
110:             $value = $property->getValue($object);
111:             if(self::isTransient($property))
112:             {
113:                 $defaults = $defaults==null?$reflection->getDefaultProperties():$defaults;
114:                 $name = $property->getName();
115:                 $property->setValue($object, array_key_exists($name, $defaults)?$defaults[$name]:null);
116:                 $altered = true;
117:             }
118:             else if(is_object($value) && $value instanceof \Closure)
119:             {
120:                 $property->setValue($object, new SerializableClosure($value));
121:                 $altered = true;
122:             }
123:             else if(is_object($value) && !($value instanceof SerializableClosure) && spl_object_hash($value)!=$hash)
124:             {
125:                 $valueReflection = new \ReflectionAnnotatedClass($value);
126:                 $altered = self::prepareForSerize($valueReflection, $value);
127:             }
128:             else if(is_array($value))
129:             {
130:                 $newValue = self::prepareArrayForSerialize($value);
131:                 
132:                 if(is_array($newValue))
133:                 {
134:                     $property->setValue($object, $newValue);
135:                     $altered = true;
136:                 }
137:             }
138:             
139:             if($altered)
140:             {
141:                 self::addRestore($property, $object, $value);
142:                 $globalAltered = true;
143:             }
144:         }
145:         
146:         $parent = $reflection->getParentClass();
147:         
148:         if($parent!=null)
149:         {
150:             self::prepareForSerize($parent, $object, true);
151:         }
152:         
153:         if(!$parent)
154:         {
155:             if($object instanceof Detachable)
156:             {
157:                 $object->detach();
158:             }
159:         }
160:         return $globalAltered;
161:     }
162:     
163:     /**
164:      * @todo This should handle array recursion
165:      * @param array $entry
166:      * @return SerializableClosure 
167:      */
168:     private static function prepareArrayForSerialize($entry)
169:     {
170:         $newEntry = array();
171:         $altered = false;
172:         
173:         foreach($entry as $key => $value)
174:         {
175:             if(is_array($value))
176:             {
177:                 $newValue = self::prepareArrayForSerialize($value);
178:                 $newEntry[$key] = is_array($newValue) ? $newValue : $value;
179:                 $altered = $altered?true:is_array($newValue);
180:             }
181:             else if(is_object($value) && $value instanceof \Closure)
182:             {
183:                 $newEntry[$key] = new SerializableClosure($value);
184:                 $altered = true;
185:             }
186:             else if(is_object($value) && !($value instanceof SerializableClosure) && !in_array(spl_object_hash($value), self::$prepared))
187:             {
188:                 $ia = self::prepareForSerize(new \ReflectionAnnotatedClass($value), $value);
189:                 $altered = $altered?true:$ia;
190:                 $newEntry[$key] = $value;
191:             }
192:             else
193:             {
194:                 $newEntry[$key] = $value;
195:             }
196:         }
197:         
198:         if($altered)
199:         {
200:             return $newEntry;
201:         }
202:         return false;
203:     }
204:     
205:     private static function addRestore(\ReflectionAnnotatedProperty $property, $object, $value)
206:     {
207:         $restore = new \stdClass();
208:         $restore->property = $property;
209:         $restore->object = $object;
210:         $restore->value = $value;
211:         self::$restore[] = $restore;
212:     }
213:     
214:     private static function restore()
215:     {
216:         foreach(self::$restore as $restore)
217:         {
218:             $restore->property->setValue($restore->object, $restore->value);
219:         }
220:     }
221:     
222:     private static function isTransient(\ReflectionAnnotatedProperty $property)
223:     {
224:         $annotations = $property->getAllAnnotations();
225:         
226:         foreach($annotations as $annotation)
227:         {
228:             return is_subclass_of($annotation, "picon\core\annotations\Transient") || get_class($annotation)=="picon\core\annotations\Transient";
229:         }
230:     }
231: }
