 27: /**
 28:  * Collection of resolvers that work as one
 29:  * 
 30:  * @author Martin Cassidy
 31:  * @package web/request/resolver
 32:  */
 33: class RequestResolverCollection implements RequestResolver
 34: {
 35:     private $resolvers = array();
 36:     
 37:     public function __construct()
 38:     {
 39:         $this->add(new PageRequestResolver());
 40:         $this->add(new PageInstanceRequestResolver());
 41:         $this->add(new ListenerRequestResolver());
 42:         $this->add(new ResourceRequestResolver);
 43:     }
 44:     
 45:     public function add(RequestResolver $resolver)
 46:     {
 47:         array_push($this->resolvers, $resolver);
 48:     }
 49:     
 50:     public function matches(Request $request)
 51:     {
 52:         foreach($this->resolvers as $resolver)
 53:         {
 54:             if($resolver->matches($request))
 55:             {
 56:                 return true;
 57:             }
 58:         }
 59:         
 60:         return false;
 61:     }
 62:     
 63:     /**
 64:      *
 65:      * @param Request $request
 66:      * @return RequestTarget the target that matches the request, null if none did 
 67:      */
 68:     public function resolve(Request $request)
 69:     {
 70:         foreach($this->resolvers as $resolver)
 71:         {
 72:             if($resolver->matches($request))
 73:             {
 74:                 return $resolver->resolve($request);
 75:             }
 76:         }
 77:         
 78:         return null;
 79:     }
 80:     
 81:     public function generateUrl(RequestTarget $target)
 82:     {
 83:         foreach($this->resolvers as $resolver)
 84:         {
 85:             if($resolver->handles($target))
 86:             {
 87:                 $url = $resolver->generateUrl($target);
 88:                 if(!empty($url))
 89:                 {
 90:                     return $url;
 91:                 }
 92:             }
 93:         }
 94:         
 95:         return null;
 96:     }
 97:     
 98:     public function handles(RequestTarget $target)
 99:     {
100:         foreach($this->resolvers as $resolver)
101:         {
102:             if($resolver->handles($target))
103:             {
104:                 return true;
105:             }
106:         }
107:         return false;
108:     }
109: }
